Cadeia de serviços multimídia com 2 vms

Esta é a implementação de uma cadeia de serviços multimídia con L-PRISM, esta cadeia compara um fluxo de vídeo original e o mesmo fluxo de vídeo processado por outra aplicação (VMS), postando finalmente o resultado em um computador (IP do computador) dentro da rede através da porta 20003.

---
chainModel:
  chainName: Example complex Chain 1 #Name of chain
  chainDescription: Description complex Chain 1 #Description of chain

  devices:
  -
    deviceName: example1 video ball #Name of device in chain
    deviceId: 638e70b10a1fbd0026fccaf4 #id of device (subcribe topic)
  -
    deviceName: complex1 color bar #Name of device in chain
    deviceId: 638e70b10a1fbd0026fccaf6 #id of device (subcribe topic)

  vmss:
  - #vms 1
    vmsName: vms1 complex1 #Name of VMS in chain
    vmsType: 638e70b10a1fbd0026fccae8 #image docker_id
    nodeIp: 192.168.0.117 #IP node deploy VMS 
    startupParameters:  #optional parameter
    virtualMemory:
      size: 1024 #optional (default docker configure))
    virtualCPU:
      numCpu: 1 #optional (default docker configure)
  - #vms 2
    vmsName: vms2 complex1 #Name of VMS in chain
    vmsType: 638e70b10a1fbd0026fccae8 #image docker_id
    nodeIp: 192.168.0.117 #IP node deploy VMS 
    startupParameters:  #optional parameter
    virtualMemory:
      size: 1024 #optional (default docker configure))
    virtualCPU:
      numCpu: 1 #optional (default docker configure)
  - #vms 3
    vmsName: vms3 complex1 #Name of VMS in chain
    vmsType: 638e70b10a1fbd0026fccae4 #image docker_id
    nodeIp: 192.168.0.117 #IP node deploy VMS 
    startupParameters:  #optional parameter
    virtualMemory:
      size: 1024 #optional (default docker configure))
    virtualCPU:
      numCpu: 1 #optional (default docker configure)

  virtualLinks:
  - #virtualLink 1
    origin:
      originName: complex1 video ball #Name of origin connect
      originType: device #Type of origin 
      PortDestine: 15000 #optional 
      outputType: video # output media stream type (vide, audi, text) 
      formatType: #optional format tipe of midia stream
    destine:
      destineName: vms2 complex1 # Destination name
      destineIp: # Configure for orchestater 
      destinePort: 15000 #information of VMS
      inputType: video # input media stream type 
  - #virtualLink 2
    origin:
      originName: complex1 video ball #Name of origin connect
      originType: device #Type of origin 
      PortDestine: 5000 #optional 
      outputType: video # output media stream type (vide, audi, text) 
      formatType: #optional format tipe of midia stream
    destine:
      destineName: vms3 complex1 # Destination name
      destineIp: # Configure for orchestater 
      destinePort: 5000 #information of VMS
      inputType: video # input media stream type 
  - #virtualLink 3
    origin:
      originName: vms3 complex1 #Name of origin connect
      originType: vms #Type of origin 
      PortDestine: 15001 #optional 
      outputType: video # output media stream type (vide, audi, text) 
      formatType: #optional format tipe of midia stream
    destine:
      destineName: vms2 complex1 # Destination name
      destineIp: # Configure for orchestater 
      destinePort: 15001 #information of VMS
      inputType: video # input media stream type 
  - #virtualLink 4
    origin:
      originName: complex1 color bar #Name of origin connect
      originType: device #Type of origin 
      PortDestine: 15001 #optional 
      outputType: video # output media stream type (vide, audi, text) 
      formatType: #optional format tipe of midia stream
    destine:
      destineName: vms1 complex1 # Destination name
      destineIp: # Configure for orchestater 
      destinePort: 15001 #information of VMS
      inputType: video # input media stream type 
  - #virtualLink 5
    origin:
      originName: vms2 complex1 #Name of origin connect
      originType: vms #Type of origin 
      PortDestine: 15000 #optional 
      outputType: video # output media stream type (vide, audi, text) 
      formatType: #optional format tipe of midia stream
    destine:
      destineName: vms1 complex1 # Destination name
      destineIp: # Configure for orchestater 
      destinePort: 15000 #information of VMS
      inputType: video # input media stream type 
  - #virtualLink 6
    origin:
      originName: vms1 complex1 #Name of origin connect
      originType: vms #Type of origin 
      PortDestine: 20005 #optional 
      outputType: video # output media stream type (vide, audi, text) 
      formatType: #optional format tipe of midia stream
    destine:
      destineName: 192.168.0.103 # Destination name

      destinePort: 20005 #Developer Information 
      inputType: video # input media stream type
